 2{
 3    if(need_sort.empty()) return;
 4
 5    for(int i=0;i<need_sort.size()-1;++i)
 6    {
 7        for(int j=0;j<need_sort.size()-i-1;++j)

插入排序

 1void insert_sort(vector<int> &ns)
 2{
 3    if(ns.empty()) return;
 4
 5    for(int i=1;i<ns.size();++i)
 6    {
 7        int key=ns[i]; // 要插入的元素
 8        int j=i-1;           // 插入的位置，暂时定为前一个
 9        while(j>=0 && ns[j]>key) // 如果前一个元素值比要插入的元素值大，则后移这个元素
10        {
11            ns[j+1]=ns[j];
12            j--;
13        }
14        ns[j+1]=key;
15    }
16}

 2{
 3    for(int i=0;i<ns.size()-1;++i)
 4    {
 5        int min=i;
 6        for(int j=i+1; j<ns.size();++j)
 7        {
 8            if(ns[j]<ns[min])
 9                min=j;
10        }
11         swap(ns[i],ns[min]);
12    }
13}

 1// 归并排序(分治和递归思想)
 2void gui_sort(vector<int> &ns,int left,int right)
 3{
 4    if(left >= right) return; //开头和结尾一样则不用排了
 5
 6    // 划分
 7    int mid=(left+right)/2;
 8    gui_sort(ns,left,mid);
 9    gui_sort(ns,mid+1,right);
10
11    // 合并
12    vector<int> tmp_vec;
13    tmp_vec.resize(ns.size());
14    int l_pos=left;
15    int r_pos=mid+1;
16    int pos=left;
17    while(l_pos<=mid && r_pos<=right)
18    {
19        if(ns[l_pos]<=ns[r_pos])
20            tmp_vec[pos++]=ns[l_pos++];
21        else
22            tmp_vec[pos++]=ns[r_pos++];
23    }
24
25    while(l_pos<=mid)
26        tmp_vec[pos++]=ns[l_pos++];
27    while(r_pos<=right)
28         tmp_vec[pos++]=ns[r_pos++];
29
30    while(left<=right)
31    {
32        ns[left]=tmp_vec[left];
33        left++;
34    }
35}

快速排序

 1// 快速排序(找一个数作为基准)
 2void quick_sort(vector<int> &ns,int left,int right)
 3{
 4    // 递归划分
 5    if(left>=right) return;
 6
 7    int pivot=ns[right];
 8    int i=left;
 9    for(int j=left;j<right;++j)
10    {
11        if(ns[j]<pivot)
12            swap(ns[j],ns[i++]);
13    }
14    swap(ns[right],ns[left]);
15    quick_sort(ns,left,i);
16    quick_sort(ns,i+1,right);
17}

堆排序

 1// 堆排序
 2//此函数维护堆，n为需要维护的总元素个数，i 为待维护的节点下标
 3void heapify(vector<int> &ns,int n,int i)
 4{
 5    // 维护堆的性质，以大顶堆为例
 6    int largest=i;
 7    int l_son=i*2+1;
 8    int r_son=2*i+2;
 9
10    if(l_son<n&&ns[l_son]>ns[largest])
11        largest=l_son;
12    if(r_son<n&&ns[largest]<ns[r_son])
13        largest=r_son;
14
15    if(largest !=i)
16    {
17        swap(ns[largest],ns[i]);
18        heapify(ns,n,largest);
19    }
20}
21
22// 建堆
23void heap_sort(vector<int> &ns)
24{
25    // 建堆
26    int n=ns.size();
27    for(int i=(n-2)/2;i>=0;--i)
28    {
29        heapify(ns,n,i);
30    }
31
32    // 排序
33    for(int i=n-1;i>0;--i)
34    {
35        swap(ns[i],ns[0]);
36        heapify(ns,i,0);
37    }
38}
