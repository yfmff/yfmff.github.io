QtUdp

发表于： 2021-12-13 分类于： qt
字数：285 阅读：≈ 1分钟 浏览：

Qt中关于UDP操作简单示例：

 1#include "widget.h"
 2#include "ui_widget.h"
 3#include<QNetworkDatagram>
 4
 5Widget::Widget(QWidget *parent)
 6    : QWidget(parent)
 7    , ui(new Ui::Widget)
 8{
 9    ui->setupUi(this);
10    setWindowTitle("端口号8000");
11    ui->btnSend->setEnabled(false);
12
13    m_socket=new QUdpSocket(this);
14    m_socket->bind(QHostAddress::Any,8000);
15
16    //    // 如果需要组播
17    //    m_socket->bind(QHostAddress::AnyIPv4,8000);
18    //    m_socket->joinMulticastGroup(QHostAddress("127.0.0.1"));
19    //    // 离开组播
20    //    m_socket->leaveMulticastGroup(QHostAddress("127.0.0.1"));
21
22    // 有数据发过来，自动触发readyread
23    connect(m_socket,&QUdpSocket::readyRead,[&]()
24    {
25        // 读数据
26        char buf[1024]={0};
27        QHostAddress tmp_addr; // 对方地址
28        quint16 port;   // 对方端口
29        m_socket->readDatagram(buf,sizeof(buf),&tmp_addr,&port);
30
31        QString txt=QString("[%1:%2]：").arg(tmp_addr.toString()).arg(port);
32        ui->editShow->append(txt);
33        ui->editShow->append(buf);
34    });
35}
36
37// 界面上有个发送消息的按钮，会发送editSend里的内容
38void Widget::on_btnSend_clicked()
39{
40    if(!m_socket) return;
41    QString ip=ui->editip->text();
42    QString port=ui->editport->text();
43    QString send_txt=ui->editSend->toPlainText();
44
45    m_socket->writeDatagram(send_txt.toUtf8(),QHostAddress(ip),port.toUInt());
46    ui->editShow->append(send_txt);
47    ui->editSend->clear();
48}
